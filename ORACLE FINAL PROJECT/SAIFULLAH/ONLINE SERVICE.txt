CREATE TABLE ORDERS
(
 ORDER_ID NUMBER(33),
 ORDER_DATE DATE,
  PRIMARY KEY(ORDER_ID)
);
CREATE TABLE PRODUCT
(
 PRODUCT_ID NUMBER(33),
 ORDER_ID NUMBER(33),
 PRICE NUMBER(33),
 PRODUCT_NAME VARCHAR2(88),
 PRIMARY KEY(PRODUCT_ID),
 FOREIGN KEY (ORDER_ID)
 REFERENCES ORDERS(ORDER_ID)
 );
CREATE TABLE COMPANY
(
COMPANY_NAME VARCHAR2(30),
COMPANY_ADDRESS VARCHAR2(80),
PRODUCT_ID NUMBER(33),
FOREIGN KEY (PRODUCT_ID)
REFERENCES PRODUCT(PRODUCT_ID)
);
CREATE TABLE PAYMENT
(
PAYMENT_ID NUMBER(33),
ORDER_ID NUMBER(33),
AMOUNT NUMBER(30),
PAYMENT_DATE DATE,
PRIMARY KEY(PAYMENT_ID),
FOREIGN KEY(ORDER_ID)
REFERENCES ORDERS(ORDER_ID) 
);
CREATE TABLE SHIPMENT
(
 SHIPMENT_ID NUMBER(33),
 ORDER_ID NUMBER(33),
 SHIPMENT_DATE DATE,
 PRIMARY KEY (SHIPMENT_ID),
 FOREIGN KEY(ORDER_ID)
 REFERENCES ORDERS(ORDER_ID)
);
CREATE TABLE CUSTOMER
(
 CUSTOMER_ID NUMBER(33),
 ORDER_ID NUMBER(33),
 CUSTOMER_NAME VARCHAR2(80),
 PHONE VARCHAR2(30),
 ADDRESS VARCHAR2(30),
 EMAIL VARCHAR2(33),
 PRIMARY KEY(CUSTOMER_ID),
 FOREIGN KEY(ORDER_ID)
 REFERENCES ORDERS(ORDER_ID)
);
