<<<PARENTS TABLE>>>
CREATE TABLE PARENTS
(
PARENT_ID NUMBER(10),
FIRST_NAME VARCHAR2(33),
LAST_NAME VARCHAR2(33),
MOBILE VARCHAR2(33),
EMAIL VARCHAR2(33) UNIQUE,
PRIMARY KEY(PARENT_ID)
);


<<<TEACHERS TABLE>>>

CREATE TABLE TEACHERS
(
TEACHER_ID NUMBER(10),
FIRST_NAME VARCHAR2(33),
LAST_NAME VARCHAR2(33),
EMAIL VARCHAR2(33) UNIQUE,
HIRE_DATE DATE,
MOBILE VARCHAR2(33),
GENDER VARCHAR2(33),
PRIMARY KEY(TEACHER_ID)
);

<<<SUBJECT TABLE>>>

CREATE TABLE SUBJECT
(
 SUBJECT_NAME VARCHAR(33),
 CLASS_NAME VARCHAR2(33),
 TEACHER_ID NUMBER(10),
 PRIMARY KEY(SUBJECT_NAME,CLASS_NAME),
 FOREIGN KEY (TEACHER_ID)
 REFERENCES TEACHERS(TEACHER_ID) 
);


<<<STUDENTS TABLE>>>

CREATE TABLE STUDENTS
(
 STUDENT_ID NUMBER(10),
 PARENT_ID NUMBER(10),
 CLASS_NAME VARCHAR2(33),
 SUBJECT_NAME VARCHAR2(33),
 FIRST_NAME VARCHAR2(33),
 LAST_NAME VARCHAR2(33),
 DATE_OF_BIRTH DATE,
 ADDMISSION_DATE DATE,
 MOBILE VARCHAR2(33),
 GENDER VARCHAR2(33),
 FATHERS_NAME VARCHAR2(33),
 ADDRESS VARCHAR2(33),
 PRIMARY KEY(STUDENT_ID),
 FOREIGN KEY(PARENT_ID)
 REFERENCES PARENTS(PARENT_ID),
 FOREIGN KEY(CLASS_NAME,SUBJECT_NAME)
 REFERENCES SUBJECT(CLASS_NAME,SUBJECT_NAME)
);

<<<NON TEACHING STAFF TABLE>>>

CREATE TABLE NON_TEACHING_STAFF
(
STAFF_ID NUMBER(10) UNIQUE,
FIRST_NAME VARCHAR2(33),
LAST_NAME VARCHAR2(33),
GENDER VARCHAR2(33),
JOB_ID VARCHAR2(33),
HIRE_DATE DATE,
MOBILE VARCHAR2(33),
ADDRESS VARCHAR2(33)
);
